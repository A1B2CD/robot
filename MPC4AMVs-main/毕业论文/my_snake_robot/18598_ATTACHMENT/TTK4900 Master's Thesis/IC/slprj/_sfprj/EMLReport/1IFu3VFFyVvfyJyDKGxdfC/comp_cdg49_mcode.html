 2function [c,d,g] = comp_cdg(J,R,theta,snake,pp,dragcoeff,zeta1)
 3%This function computes the terms c=C(theta,zeta1)zeta2,
 4%d=D(theta,zeta1)zeta2, and g(eta,theta), i.e. the coriolis, drag and
 5%hydrostatic forces on the snake. The sign-convention is such that these
 6%enter the equation of motion on the left-hand side, i.e. the actual force
 7%in terms of newton's laws is what this script puts out times -1.
 8
 9
10nu1 = zeta1(1:6);
11nu2=nu1;
12
13c  = -J(:,:,1)'*W(snake.link(1).M*nu1)*nu2;
14xi = zeros(6,1);
15    if snake.link(1).dragtype==1
16        d  = J(:,:,1)'*zeros(6,1);
17    else
18        d  = J(:,:,1)'*drag(nu1,nu2,snake.link(1).length,pp.denw,snake.r,dragcoeff);
19    end
20
21    g  = J(:,:,1)'*snake.link(1).G*R(:,:,1)'*pp.gdir;
22
23
24 for i=2:snake.n
25     j=i-1;
26    if snake.joint(j).type==1
27        A=[Rotx(theta(j)), [snake.link(i).length 0 0]'; [0 0 0 1]];
28    elseif snake.joint(j).type==2
29        A=[Roty(theta(j)), [snake.link(i).length 0 0]'; [0 0 0 1]];
30    else
31        A=[Rotz(theta(j)), [snake.link(i).length 0 0]'; [0 0 0 1]];
32    end           
33     twist    = snake.joint(j).twist;            
34     Adinv    = Ad_inv(A);                       
35     ad_tw    = ad(twist); 
36     nu1      = Adinv*nu1 + twist*zeta1(6+j);
37     nu2   = nu1;
38     xi       = -ad_tw*zeta1(6+j)*nu2 + Adinv*xi;
39     M        = snake.link(i).M;
40     
41     c        = c + J(:,:,i)'*(M*xi - W(M*nu1)*nu2);
42     if snake.link(i).dragtype==1
43          d   = d + J(:,:,i)'*zeros(6,1);
44     else
45          d   = d + J(:,:,i)'*drag(nu1,nu2,snake.link(i).length,pp.denw,snake.r,dragcoeff);  %Sjekk om det er riktig å ha i her
46     end
47          g   = g + J(:,:,i)'*snake.link(i).G*R(:,:,i)'*pp.gdir;
48 end
49end
