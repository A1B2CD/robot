1function [x_dot, zeta_dot,theta] = dyn(x,joint_torque,thrust , pp, snake, dragcoeff)
 2
 3pos       =  x(1:3);
 4quat      =  x(4:7);
 5theta     =  x(8 : 6+snake.n);
 6zeta      =  x(7+snake.n : 11+2*snake.n);
 7%Compute rotation matrices and jacobians
 8[Rcdg,Jcdg]=comp_rot_jac_mcdg(quat,theta,snake);
 9%Compute mass matrix
10M         =  comp_mass(Jcdg,snake);
11
12%Compute relative velocity to take into account the effect of current
13velc_h    =  Rcdg(:,:,1)'*pp.velc;
14nu        =  zeta(1:6);
15zeta(1:3) =  zeta(1:3)-velc_h;
16
17%Compute coriolis, drag, and hydrostatic terms with the relative velocity
18[c, d, g] =  comp_cdg(Jcdg,Rcdg,theta,snake,pp,dragcoeff,zeta);
19%Compute the thrust-configuration matrix
20B         =  comp_TCM(Jcdg,snake);
21
22
23%Compute the relative acceleration
24zeta_dot      =  M\([B [zeros(6,8); eye(8)]]*[thrust;joint_torque]-c-d-g);% -c-d-g
25
26%Remove the effect of current
27zeta_dot(1:3) = zeta_dot(1:3)-cross(nu(4:6),velc_h);
28%Compute the time derivatives of position and orientation
29pos_dot   =  Rcdg(:,:,1)*nu(1:3);  
30quat_dot  =  TransAngQ(quat)*nu(4:6)+1*(1-quat'*quat)*quat; 
31%Assemble derivative of the state vector
32x_dot = [pos_dot ; quat_dot ; zeta(7:snake.n+5) ; zeta_dot];
33end
