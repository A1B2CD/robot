1function out = comp_TCM(J,snake)
 2%This function computes the thrust-configuration matrix for the robot. 
 3out          = zeros(5+snake.n,snake.n_thruster_tot); 
 4m            = snake.link(1).n_thruster;
 5out(:,1:m)   = J(:,:,1)'*snake.link(1).B; 
 6
 7 for i = 2:snake.n
 8     if snake.link(i).n_thruster > 0
 9        m = m + snake.link(i).n_thruster;
10        out(:,m+1-snake.link(i).n_thruster:m) = J(:,:,i)'*snake.link(i).B;
11    end
12 end
13end
