1function [R,J] = comp_rot_jac(quat,theta,snake)
 2%This function computes the rotation matrices and body-jacobians for the
 3%snake robot.
 4rot = zeros(3,3,snake.n+1);
 5rot(:,:,1) = RotQ(quat);
 6
 7
 8jac = zeros(6,5+snake.n,snake.n+1);
 9jac(1:6,1:6,1) = eye(6);
10
11
12 for i = 2:snake.n
13    j              = i-1;     
14    if snake.joint(j).type==1
15        A=[Rotx(theta(j)), [snake.link(j).length 0 0]'; [0 0 0 1]];
16    elseif snake.joint(j).type==2
17        A=[Roty(theta(j)), [snake.link(j).length 0 0]'; [0 0 0 1]];
18    else
19        A=[Rotz(theta(j)), [snake.link(j).length 0 0]'; [0 0 0 1]];
20    end           
21    rot(:,:,i)     = rot(:,:,i-1)*A(1:3,1:3);
22    
23    jac(:,:,i)   = Ad_inv(A)*jac(:,:,i-1);
24    jac(:,6+j,i) = jac(:,6+j,i)+snake.joint(j).twist;
25 end
26 
27     A=[Rotz(0), [snake.link(snake.n).length 0 0]'; [0 0 0 1]];
28    jac(:,:,snake.n+1)   = Ad_inv(A)*jac(:,:,snake.n);
29    rot(:,:,snake.n+1)     = rot(:,:,snake.n)*A(1:3,1:3);
30 
31     R = rot;
32     J = jac;
33
34end
