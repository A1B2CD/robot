1function out = drag(nu1,nu2,l,rho,r,dragcoeff)
 2%This function computes the drag wrench D(nu1)nu2 for a link.
 3slices = 10;
 4x   = linspace(0,1,slices);
 5dx  = x(2)-x(1);
 6d_N = zeros(6,1);
 7d_L = d_N;
 8
 9v_ref = 1;
10
11C_d_1 = dragcoeff(1);
12C_d_4 = dragcoeff(2);
13C_d_C = dragcoeff(3);
14
15C_d_L = dragcoeff(4);
16beta  = dragcoeff(5);
17gamma = dragcoeff(6);
18
19d_N(1)  = .5*rho*pi*r^2*C_d_1*abs(nu1(1))*nu2(1); 
20d_N(4)  = .5*rho*pi*l*r^4*C_d_4*abs(nu1(4))*nu2(4);
21
22sqrtx   =  sqrt(nu1(2)^2+nu1(3)^2+x.^2*nu1(5)^2+x.^2*nu1(6)^2+2*x*nu1(2)*nu1(6)-2*x*nu1(3)*nu1(5));
23
24k_N     =  rho*r*dx*C_d_C*l;
25d_N(2)  =  k_N*trapz(sqrtx.*(nu2(2)+nu2(6)*x));
26d_N(3)  =  k_N*trapz(sqrtx.*(nu2(3)-nu2(5)*x));
27d_N(5)  =  k_N*trapz(sqrtx.*(-nu2(3)*x+nu2(5)*x.^2));
28d_N(6)  =  k_N*trapz(sqrtx.*(nu2(2)*x+nu2(6)*x.^2));
29
30k_L     = rho*r*l*C_d_L*v_ref;
31d_L(1)  = k_L*beta*nu2(1);
32d_L(2)  = k_L*(nu2(2)+0.5*l*nu2(6));
33d_L(3)  = k_L*(nu2(3)-0.5*l*nu2(5));
34d_L(4)  = k_L*gamma*r^2*nu2(4);
35d_L(5)  = k_L*(-0.5*l*nu2(3)+l^2*nu2(5)/3);
36d_L(6)  = k_L*(0.5*l*nu2(2)+l^2*nu2(6)/3);
37
38out= d_N + d_L;
39end
